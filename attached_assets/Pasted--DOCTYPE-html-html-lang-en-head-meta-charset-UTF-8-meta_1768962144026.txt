  <script>
    const form = document.getElementById('form');
    const btn = document.getElementById('btn');
    const errorDiv = document.getElementById('error');
    const resultDiv = document.getElementById('result');
    const linkInput = document.getElementById('link');
    const copyBtn = document.getElementById('copy');
    const copiedDiv = document.getElementById('copied');
    const planSelect = document.getElementById('plan');

    form.addEventListener('submit', async (e) => {
      e.preventDefault();
      if (!document.getElementById("agree").checked) {
        errorDiv.textContent = "You must agree to the terms before continuing.";
        return;
      }
      errorDiv.textContent = '';
      resultDiv.classList.remove('show');
      copiedDiv.textContent = '';

      btn.disabled = true;
      btn.textContent = 'Generating...';

      try {
        const res = await fetch('/api/create-link', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({ plan: planSelect.value })

        });

        const data = await res.json();
        
        if (!res.ok) {
          throw new Error(data.error || 'Failed to create link');
        }

        linkInput.value = data.private_url;
        resultDiv.classList.add('show');
      } catch (err) {
        errorDiv.textContent = err.message;
      } finally {
        btn.disabled = false;
        btn.textContent = 'Generate Payment Link';
      }
    });

    copyBtn.addEventListener('click', async () => {
      await navigator.clipboard.writeText(linkInput.value);
      copiedDiv.textContent = 'Copied to clipboard!';
      setTimeout(() => copiedDiv.textContent = '', 2000);
    });
 </script>
  
  <script>
    async function generateLink() {
      const plan = document.getElementById("plan").value;

      const res = await fetch("/api/create-link", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ plan })
      });

      const data = await res.json();
      if (data.private_url) {
        window.location.href = data.private_url;
      }
    }
